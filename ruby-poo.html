Ruby: Programação Orientada a Objetos

Esse programa cria uma classe Animal com duas classes filhas: Pessoa e Cachorro. PESSOA E CACHORRO!

class Animal
end

class Pessoa < Animal
  attr_accessor :name

  def initialize(nome)
    self.name = nome
  end

  def passearComCachorro(cachorro)
    puts "#{self.name} está passeando com #{cachorro.name}."
  end
end

class Cachorro < Animal
  attr_accessor :name

  def initialize(nome)
    self.name = nome
  end
end

nedio = Pessoa.new("Nédio")
doguinho = Cachorro.new("Doguinho")

nedio.passearComCachorro(doguinho)
