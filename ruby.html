Fibonacci em Ruby

Esse é um programa simples em Ruby que imprime n números da sequência de Fibonacci:


def fibonacci(n)
    if (n < 1)
        return
    end

    ant = 1
    fib = 0

    for i in 1..n
        fib = fib + ant
        ant = fib - ant
        puts "#{fib}"
    end
end

puts "Quantos numeros da sequencia voce quer computar?"
numero = gets.to_i
fibonacci(numero)
